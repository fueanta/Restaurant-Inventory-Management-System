CREATE table "FOOD_INGREDIENT" (
    "FOOD_ID"       NUMBER(3,0),
    "INGREDIENT_ID" NUMBER(3,0),
    "AMOUNT"        NUMBER(10,4) NOT NULL
)
/

alter table "FOOD_INGREDIENT" add constraint  "FOOD_INGREDIENT_PK" primary key ("FOOD_ID","INGREDIENT_ID")
/

ALTER TABLE "FOOD_INGREDIENT" ADD CONSTRAINT "FOOD_INGREDIENT_FK" 
FOREIGN KEY ("FOOD_ID")
REFERENCES "FOOD" ("ID")
ON DELETE CASCADE

/
ALTER TABLE "FOOD_INGREDIENT" ADD CONSTRAINT "FOOD_INGREDIENT_FK2" 
FOREIGN KEY ("INGREDIENT_ID")
REFERENCES "INGREDIENT" ("ID")
ON DELETE SET NULL

/

constraint "FOOD_INGREDIENT_CK1"
